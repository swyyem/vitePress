Redisе…Ҙй—ЁеҸҠе®һи·ө вҖӢ

д»Җд№ҲжҳҜRedisпјҹ вҖӢ

RedisпјҲRemote Dictionary ServerпјүжҳҜдёҖдёӘејҖжәҗзҡ„гҖҒй«ҳжҖ§иғҪзҡ„гҖҒеҹәдәҺеҶ…еӯҳзҡ„й”®еҖјж•°жҚ®еӯҳеӮЁзі»з»ҹгҖӮе®ғеҸҜд»Ҙз”ЁдҪңж•°жҚ®еә“гҖҒзј“еӯҳгҖҒж¶ҲжҒҜдёӯй—ҙд»¶зӯүеӨҡз§Қз”ЁйҖ”гҖӮRedisж”ҜжҢҒеӨҡз§Қж•°жҚ®зұ»еһӢпјҢеҢ…жӢ¬еӯ—з¬ҰдёІпјҲStringпјүгҖҒе“ҲеёҢпјҲHashпјүгҖҒеҲ—иЎЁпјҲListпјүгҖҒйӣҶеҗҲпјҲSetпјүе’ҢжңүеәҸйӣҶеҗҲпјҲSorted SetпјүзӯүгҖӮ

Redisзҡ„дё»иҰҒзү№зӮ№пјҡ вҖӢ

  1. й«ҳжҖ§иғҪпјҡRedisе°ҶжүҖжңүж•°жҚ®еӯҳеӮЁеңЁеҶ…еӯҳдёӯпјҢеӣ жӯӨеҸҜд»Ҙе®һзҺ°жһҒеҝ«зҡ„иҜ»еҶҷйҖҹеәҰ
  2. жҢҒд№…еҢ–пјҡж”ҜжҢҒж•°жҚ®зҡ„жҢҒд№…еҢ–пјҢеҸҜд»Ҙе°ҶеҶ…еӯҳдёӯзҡ„ж•°жҚ®дҝқеӯҳеҲ°зЈҒзӣҳдёӯ
  3. еҺҹеӯҗжҖ§пјҡRedisзҡ„жүҖжңүж“ҚдҪңйғҪжҳҜеҺҹеӯҗжҖ§зҡ„пјҢж”ҜжҢҒдәӢеҠЎ
  4. еӨҡз§Қж•°жҚ®зұ»еһӢпјҡж”ҜжҢҒеӨҡз§Қж•°жҚ®з»“жһ„пјҢж»Ўи¶ідёҚеҗҢеңәжҷҜйңҖжұӮ
  5. дё»д»ҺеӨҚеҲ¶пјҡж”ҜжҢҒж•°жҚ®зҡ„еӨҮд»Ҫе’ҢиҜ»еҶҷеҲҶзҰ»
  6. й«ҳеҸҜз”ЁпјҡйҖҡиҝҮRedis Sentinelе’ҢRedis ClusterдҝқиҜҒжңҚеҠЎзҡ„й«ҳеҸҜз”ЁжҖ§

Redisзҡ„еҠҹиғҪе’ҢдҪңз”Ё вҖӢ

1. зј“еӯҳзі»з»ҹ вҖӢ

  • еҮҸиҪ»ж•°жҚ®еә“еҺӢеҠӣ
  • жҸҗй«ҳе“Қеә”йҖҹеәҰ
  • ж”ҜжҢҒж•°жҚ®иҝҮжңҹзӯ–з•Ҙ

2. и®Ўж•°еҷЁ вҖӢ

  • е®һж—¶з»ҹи®Ў
  • и®ҝй—®йҮҸз»ҹи®Ў
  • зӮ№иөһж•°з»ҹи®Ў

3. йҳҹеҲ—зі»з»ҹ вҖӢ

  • ж¶ҲжҒҜйҳҹеҲ—
  • е»¶иҝҹйҳҹеҲ—
  • д»»еҠЎйҳҹеҲ—

4. дјҡиҜқз®ЎзҗҶ вҖӢ

  • еӯҳеӮЁз”ЁжҲ·session
  • з”ЁжҲ·tokenз®ЎзҗҶ
  • еңЁзәҝз”ЁжҲ·з®ЎзҗҶ

5. жҺ’иЎҢжҰң вҖӢ

  • е®һж—¶жҰңеҚ•
  • з§ҜеҲҶжҺ’еҗҚ
  • зғӯй—Ёе•Ҷе“Ғ

еҹәдәҺNest.jsйЎ№зӣ®дёӯзҡ„Redisе®һи·ө вҖӢ

дёӢйқўжҲ‘д»¬е°ҶйҖҡиҝҮдёҖдёӘе®һйҷ…жЎҲдҫӢпјҢеұ•зӨәеҰӮдҪ•еңЁNest.jsйЎ№зӣ®дёӯдҪҝз”ЁRedisиҝӣиЎҢжҖ§иғҪдјҳеҢ–гҖӮ

е®һзҺ°зј“еӯҳжңҚеҠЎ вҖӢ

еҲӣе»әдёҖдёӘз”ЁжҲ·жңҚеҠЎдҪңдёәзӨәдҫӢпјҡ

typescript
import { Injectable, Inject } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';

@Injectable()
export class UserService {
  constructor(
    @Inject(CACHE_MANAGER) private cacheManager: Cache
  ) {}

  async getUserById(id: number) {
    // е°қиҜ•д»Һзј“еӯҳиҺ·еҸ–ж•°жҚ®
    const cachedUser = await this.cacheManager.get(`user:${id}`);
    if (cachedUser) {
      return cachedUser;
    }

    // жЁЎжӢҹд»Һж•°жҚ®еә“иҺ·еҸ–ж•°жҚ®
    const user = await this.findUserFromDB(id);
    
    // е°Ҷж•°жҚ®еӯҳе…Ҙзј“еӯҳ
    await this.cacheManager.set(`user:${id}`, user, 3600);
    
    return user;
  }

  private async findUserFromDB(id: number) {
    // жЁЎжӢҹж•°жҚ®еә“жҹҘиҜў
    return {
      id,
      name: 'John Doe',
      email: 'john@example.com'
    };
  }
}

import { Injectable, UseInterceptors, CacheInterceptor } from '@nestjs/common';

      { id: 1, name: 'Product 1' },
      { id: 2, name: 'Product 2' }
    ];
  }
}

жҖ§иғҪдјҳеҢ–ж•Ҳжһң вҖӢ

йҖҡиҝҮдҪҝз”ЁRedisзј“еӯҳпјҢжҲ‘д»¬еҸҜд»Ҙе®һзҺ°д»ҘдёӢдјҳеҢ–ж•Ҳжһңпјҡ

  1. е“Қеә”йҖҹеәҰжҸҗеҚҮпјҡйҰ–ж¬ЎиҜ·жұӮеҗҺзҡ„ж•°жҚ®е°ҶзӣҙжҺҘд»ҺRedisиҝ”еӣһпјҢйҒҝе…ҚдәҶж•°жҚ®еә“жҹҘиҜў
  2. ж•°жҚ®еә“еҺӢеҠӣйҷҚдҪҺпјҡеҮҸе°‘дәҶйҮҚеӨҚзҡ„ж•°жҚ®еә“жҹҘиҜўиҜ·жұӮ
  3. й«ҳе№¶еҸ‘ж”ҜжҢҒпјҡRedisзҡ„й«ҳжҖ§иғҪзү№жҖ§дҪҝзі»з»ҹиғҪжӣҙеҘҪең°еӨ„зҗҶе№¶еҸ‘иҜ·жұӮ

жіЁж„ҸдәӢйЎ№ вҖӢ

  1. зј“еӯҳжӣҙж–°зӯ–з•Ҙпјҡ

    • еҸҠж—¶жӣҙж–°жҲ–жё…йҷӨиҝҮжңҹзј“еӯҳ
    • иҖғиҷ‘зј“еӯҳйў„зғӯ
    • е®һзҺ°зј“еӯҳеӨұж•ҲжңәеҲ¶

  2. ж•°жҚ®дёҖиҮҙжҖ§пјҡ

    • зЎ®дҝқзј“еӯҳдёҺж•°жҚ®еә“зҡ„ж•°жҚ®еҗҢжӯҘ
    • еҗҲзҗҶи®ҫзҪ®зј“еӯҳиҝҮжңҹж—¶й—ҙ
    • еңЁж•°жҚ®жӣҙж–°ж—¶еҸҠж—¶жё…йҷӨзӣёе…ізј“еӯҳ

  3. еҶ…еӯҳз®ЎзҗҶпјҡ

    • еҗҲзҗҶи®ҫзҪ®зј“еӯҳеӨ§е°Ҹ
    • е®һзҺ°зј“еӯҳж·ҳжұ°зӯ–з•Ҙ
    • зӣ‘жҺ§RedisеҶ…еӯҳдҪҝз”Ёжғ…еҶө

жҖ»з»“ вҖӢ

RedisдҪңдёәдёҖдёӘй«ҳжҖ§иғҪзҡ„зј“еӯҳзі»з»ҹпјҢеңЁзҺ°д»Јwebеә”з”Ёдёӯжү®жј”зқҖи¶ҠжқҘи¶ҠйҮҚиҰҒзҡ„и§’иүІгҖӮйҖҡиҝҮеңЁNest.jsйЎ№зӣ®дёӯйӣҶжҲҗRedisпјҢжҲ‘д»¬еҸҜд»Ҙжҳҫи‘—жҸҗеҚҮеә”з”ЁжҖ§иғҪпјҢдјҳеҢ–з”ЁжҲ·дҪ“йӘҢгҖӮеңЁе®һйҷ…дҪҝз”ЁдёӯпјҢйңҖиҰҒж №жҚ®е…·дҪ“дёҡеҠЎеңәжҷҜйҖүжӢ©еҗҲйҖӮзҡ„зј“еӯҳзӯ–з•ҘпјҢеҗҢж—¶жіЁж„Ҹж•°жҚ®дёҖиҮҙжҖ§е’ҢеҶ…еӯҳз®ЎзҗҶзӯүй—®йўҳгҖӮ
